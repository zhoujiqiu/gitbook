pullup

上拉加载更多数据

引入

  import pullup from 'toon-ui/lib/components/Pullup'
  components: {
    pullup
  }

例子

  <pullup :bottom-method="loadBottom" :bottom-status.sync="bottomStatus" :bottom-all-loaded="allLoaded">
    <ul>
      <li v-for="item in list">{{ item }}</li>
    </ul>
  </pullup>

参数配置

  list: [],
  allLoaded: false,
  bottomStatus: ''
